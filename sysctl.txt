cat <<'EOF' | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF
sudo modprobe overlay
sudo modprobe br_netfilter

# 7) Sysctl for k8s networking
cat <<'EOF' | sudo tee /etc/sysctl.d/99-kubernetes-cri.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF

sudo sysctl --system

# 8) Ensure container-selinux is present (needed for Podman/CRI-O on SELinux systems)
# If this fails, enable the proper RHEL repos or install from the vendor-provided package.

sudo dnf -y install container-selinux
